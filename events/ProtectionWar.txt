	id = protectionwar.1
	title = protectionwar.1.t
	desc = protectionwar.1.d

		date > 1915.12.01
		original_tag = QIN
		has_government = democratic

		name = protectionwar.1.o2

		ai_chance = { factor = 10 }
		add_popularity = { ideology = fascism popularity = 0.3 }
		country_event = { days = 13 id = protectionwar.4 }
	}
}
